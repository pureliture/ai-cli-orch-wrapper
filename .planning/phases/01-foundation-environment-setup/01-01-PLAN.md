  function isOnPath(tool: string): boolean {
    const result = spawnSync('which', [tool], { encoding: 'utf8' });
    return result.status === 0;
  }

Main exported function:

  export async function setupCommand(): Promise&lt;void&gt; {
    // Step 1: Prerequisite check (per D-04, D-05, D-06)
    const missing = REQUIRED_TOOLS.filter(t =&gt; !isOnPath(t));
    if (missing.length &gt; 0) {
      console.error(`Error: missing prerequisites: ${missing.join(', ')}`);
      process.exit(1);
    }
    console.log('[checkmark] prerequisites: cao, tmux, workmux found');

    // Step 2: Write ~/.config/tmux/ai-cli.conf (per D-07, D-08)
    if (existsSync(AI_CLI_CONF)) {
      console.log('[checkmark] ~/.config/tmux/ai-cli.conf: already exists');
    } else {
      mkdirSync(AI_CLI_CONF_DIR, { recursive: true });
      writeFileSync(AI_CLI_CONF, AI_CLI_CONF_CONTENT, 'utf8');
      console.log('[checkmark] ~/.config/tmux/ai-cli.conf written');
    }

    // Step 3: Inject source-file line into ~/.tmux.conf (per D-07, D-09)
    if (!existsSync(TMUX_CONF)) {
      writeFileSync(TMUX_CONF, SOURCE_LINE + '\n', 'utf8');
      console.log('[checkmark] ~/.tmux.conf: source line added');
    } else {
      const content = readFileSync(TMUX_CONF, 'utf8');
      if (content.includes(AI_CLI_CONF)) {
        console.log('[checkmark] ~/.tmux.conf: already configured');
      } else {
        appendFileSync(TMUX_CONF, '\n' + SOURCE_LINE + '\n', 'utf8');
        console.log('[checkmark] ~/.tmux.conf: source line added');
      }
    }

    console.log('Setup complete.');
  }

IMPORTANT: Replace the literal text "[checkmark]" in console.log calls with the actual Unicode checkmark character [✓] (U+2713). The action text above uses the word to avoid encoding issues in the plan document.
