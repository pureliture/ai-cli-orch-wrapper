```bash
# aco_adapter_available <key>
# Returns 0 if binary in PATH, 1 otherwise.
# Does NOT print anything — caller handles display.
# Example: if aco_adapter_available "gemini"; then echo "✓ gemini ..."; fi

# aco_adapter_version <key>
# Prints version string to stdout.
# Returns "unavailable" if binary not in PATH (never crashes).
# Example: VERSION=$(aco_adapter_version "gemini")
#   gemini installed:     "0.36.0" (or similar)
#   gemini not installed: "unavailable"
```
